{% macro renderNavigation(navigation) %}
<ul>
  {% for item in navigation %}
  <li {{ item.attr|safe_html }} data-depth="c_{{ item.depth }}" >
    <a href="{{ item.url }}" {{ item.attr2|safe_html }} target="{{ item.target }}">{{ item.label }}</a>
    {% if item.children is defined %}
      {{ _self.renderNavigation(item.children) }}
    {% endif %}
  </li>
  {% endfor %}
</ul>
{% endmacro %}

  <nav class="acms-navbar acms-margin-bottom-medium clearfix">
    {{ _self.renderNavigation(navigation.items) }}
  </nav>
